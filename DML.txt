✅ 1. Create Table (Employee Details)
CREATE TABLE employee (
    emp_id INT PRIMARY KEY,
    name VARCHAR(50),
    department VARCHAR(50),
    salary INT
);
✅ 2. Create Stored Procedure with INPUT Parameters
DELIMITER //

CREATE PROCEDURE AddEmployee(
    IN e_id INT,
    IN e_name VARCHAR(50),
    IN e_dept VARCHAR(50),
    IN e_salary INT
)
BEGIN
    INSERT INTO employee(emp_id, name, department, salary)
    VALUES (e_id, e_name, e_dept, e_salary);
END //

DELIMITER ;
✅ 3. Call the Procedure (Pass Parameters)
CALL AddEmployee(1, 'Rahul', 'IT', 30000);
CALL AddEmployee(2, 'Sneha', 'HR', 35000);
✅ 4. Verify Data
SELECT * FROM employee;
