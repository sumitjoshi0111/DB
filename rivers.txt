a) Number of jobs available in employees table
SELECT COUNT(DISTINCT JOB_ID) AS NO_OF_JOBS
FROM EMPLOYEES;
b) Total salaries payable to employees
SELECT SUM(SALARY) AS TOTAL_SALARY
FROM EMPLOYEES;
c) Minimum salary from employees table
SELECT MIN(SALARY) AS MIN_SALARY
FROM EMPLOYEES;
d) Maximum salary of an employee working as a Programmer
SELECT MAX(SALARY) AS MAX_PROGRAMMER_SALARY
FROM EMPLOYEES
WHERE JOB_ID = 'IT_PROG';

(Note: In Oracle HR schema, Programmer job_id is usually IT_PROG)

e) Average salary and number of employees in department 90
SELECT 
    AVG(SALARY) AS AVG_SALARY,
    COUNT(*) AS NO_OF_EMPLOYEES
FROM EMPLOYEES
WHERE DEPARTMENT_ID = 90;

1) Create table RIVERS
name → should NOT be NULL and UNIQUE (not repeated)
length → numeric
outflow → initially defined (you can modify later)
CREATE TABLE RIVERS (
    name VARCHAR2(50) NOT NULL UNIQUE,
    length NUMBER,
    outflow VARCHAR2(50)
);
2) Add a new column MaxDepth
ALTER TABLE RIVERS
ADD MaxDepth NUMBER;
3) Change datatype of outflow column

(Example: changing from VARCHAR2 to NUMBER or increasing size)

👉 If changing to a larger VARCHAR:

ALTER TABLE RIVERS
MODIFY outflow VARCHAR2(100);

👉 If changing datatype completely (e.g., to NUMBER):

ALTER TABLE RIVERS
MODIFY outflow NUMBER;
