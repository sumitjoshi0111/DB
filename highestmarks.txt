✅ Create STUDENT Table
CREATE TABLE STUDENT (
    id NUMBER PRIMARY KEY,
    name VARCHAR2(50),
    math_marks NUMBER,
    dbms_marks NUMBER,
    attendance NUMBER   -- in percentage
);

✅ 1) Student(s) having Highest Marks in Math
SELECT *
FROM STUDENT
WHERE math_marks = (SELECT MAX(math_marks) FROM STUDENT);
✅ 2) Student(s) having Lowest Attendance
SELECT *
FROM STUDENT
WHERE attendance = (SELECT MIN(attendance) FROM STUDENT);
✅ 3) Total Number of Students
SELECT COUNT(*) AS total_students
FROM STUDENT;
✅ 4) Average Marks of DBMS
SELECT AVG(dbms_marks) AS avg_dbms_marks
FROM STUDENT;
