✅ 1. Assume Customer Table
CREATE TABLE Customer (
    cust_id INT PRIMARY KEY,
    name VARCHAR2(50),
    salary NUMBER
);
✅ 2. Implicit Cursor Example (UPDATE + ROWCOUNT)
SET SERVEROUTPUT ON;

BEGIN
    -- Update salary of all customers
    UPDATE Customer
    SET salary = salary + 5000;

    -- Display number of rows affected
    DBMS_OUTPUT.PUT_LINE('Rows Updated: ' || SQL%ROWCOUNT);
END;

SELECT *
FROM salespeople
WHERE city = 'New York';
🔹 Check View

✅ 4. (b) View to Count Salespeople in Each City
CREATE VIEW City_Count AS
SELECT city, COUNT(*) AS total_salespeople
FROM salespeople
GROUP BY city;
